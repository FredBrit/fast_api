### SELECT

from sqlalchemy import select, func
from src.models import HotelsORM

### INSERT

# 1. Добавление одной записи
new_hotel = HotelsORM(title="Новый отель", location="Сочи")
session.add(new_hotel)
await session.commit()
await session.refresh(new_hotel) # Чтобы получить ID и значения по умолчанию из БД

# 2. Добавление нескольких записей сразу
hotels = [
    HotelsORM(title="Отель 1", location="Москва"),
    HotelsORM(title="Отель 2", location="Казань")
]
session.add_all(hotels)
await session.commit()


### UPDATE 
# Способ А: Через ORM (рекомендуется для сложной логики)
hotel = await session.get(HotelsORM, 1)
if hotel:
    hotel.title = "Обновленное название"
    await session.commit()

# Способ Б: Массовое обновление (Bulk Update, SQLAlchemy 2.0)
# Быстрее, но не загружает объекты в память и не триггерит события ORM
from sqlalchemy import update
stmt = (
    update(HotelsORM)
    .where(HotelsORM.location == "Москва")
    .values(title="Отель в Москве") # .values() принимает словарь или именованные аргументы
)
await session.execute(stmt)
await session.commit()


### DELETE
# Способ А: Через ORM
hotel = await session.get(HotelsORM, 1)
if hotel:
    await session.delete(hotel)
    await session.commit()

# Способ Б: Массовое удаление (Bulk Delete)
from sqlalchemy import delete
stmt = delete(HotelsORM).where(HotelsORM.location == "Сочи")
await session.execute(stmt)
await session.commit()
