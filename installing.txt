### Подгружаем системные зависимости

# RHEL
dnf install python3-devel libpq-devel python3-virtualenv -y

# Debian
apt install python3-pip python3-venv python3-dev libpq-dev -y

### Создание venv (лучше создавать в /opt)
mkdir my_fastapi_app && cd my_fastapi_app
python3 -m venv venv
source venv/bin/activate
pip install --upgrade pip

### Подгружем необходимые для асинхронной работы библиотеки (находясь внутри активированной venv)
pip install \
  "fastapi[standard]" \
  pydantic \
  pydantic-settings \
  pandas \
  asyncpg \
  sqlalchemy[asyncio] \
  alembic \
  httpx \
  python-dotenv \
  uvicorn[standard]


### Создаем requirements.txt
pip freeze > requirements.txt


### Структуру классического FastAPI проекта
my_fastapi_app/
├── .env                  # пароли, URL БД (НЕ коммитить в git!)
├── .gitignore            # обязательно добавить venv/ и .env сюда!
├── requirements.txt      # список зависимостей (pip freeze > requirements.txt)
├── main.py               # точка входа, создание FastAPI()
│
├── venv/                 # 📦 ВИРТУАЛЬНОЕ ОКРУЖЕНИЕ (создается командой python3 -m venv venv)
│   ├── bin/              # ⚙️ Исполняемые файлы: python, pip, uvicorn и скрипт activate
│   ├── include/          # 🧠 Заголовочные файлы C (нужны для компиляции пакетов типа psycopg2)
│   ├── lib/              # 📚 Основные библиотеки
│   │   └── python3.X/    # Папка с версией вашего Python (например, python3.9)
│   │       └── site-packages/  # 🎯 САМОЕ ГЛАВНОЕ: здесь лежат все установленные вами пакеты 
│   │                           # (fastapi, uvicorn, sqlalchemy, pydantic и т.д.)
│   ├── lib64/            # Симлинк на lib (есть не во всех ОС)
│   └── pyvenv.cfg        # ⚙️ Конфигурация самого виртуального окружения (путь к системному Python)
│
├── app/
│   ├── __init__.py
│   ├── config.py         # pydantic-settings (чтение .env)
│   ├── database.py       # engine, sessionmaker
│   ├── models.py         # SQLAlchemy модели
│   ├── schemas.py        # Pydantic схемы
│   ├── routers/          # эндпоинты по модулям
│   └── services/         # бизнес-логика
│
└── alembic/              # миграции
    ├── env.py
    ├── script.py.mako
    └── versions/         # файлы миграций (001_init.py и т.д.)

### Что вам нужно знать о venv:
Вам не нужно вручную создавать или класть туда файлы.
Вся магия происходит через pip. Когда вы делаете pip install fastapi, пакет физически скачивается и распаковывается в папку venv/lib/python3.X/site-packages/.
Папка bin/ — это ваш новый "центр управления".
Когда окружение активировано, команды python, pip и uvicorn берутся именно из venv/bin/, а не из системных папок.
Никогда не коммитьте venv в Git!
Папка venv весит много, зависит от операционной системы и версии Python. В Git хранится только requirements.txt. 
Другой разработчик (или вы на другом сервере) просто скачает код и создаст свой venv командой python3 -m venv venv, а затем установит пакеты из requirements.txt.


### Запуск проекта
cd my_fastapi_app
source venv/bin/activate (редактировать файлы можно БЕЗ активации venv)

Переходим в директорию, активириуем venv
Далее запуск веб-сервера uvicorn с ключом --reload (автоматически перезагружать сервер при изменении кода)
uvicorn main:app --host 0.0.0.0 --port 8000 --reload

Проверка работы
http://your-server-ip:8000/ — увидите {"message": "Hello World"}
http://your-server-ip:8000/docs — автоматическая Swagger документация

### Выделение проекта в отдельный постоянно работающий сервис без необходиомости удерживать подключение по ssh через терминал
sudo nano /etc/systemd/system/myfastapi.service
Пример текста в файле:
[Unit]
Description=My FastAPI Application
After=network.target

[Service]
User=admin
Group=admin
WorkingDirectory=/home/my_fastapi_app
Environment="PATH=/home/my_fastapi_app/venv/bin"
# Если нужно передать .env, можно добавить: EnvironmentFile=/home/my_fastapi_app/.env
ExecStart=/home/my_fastapi_app/venv/bin/uvicorn main:app --host 0.0.0.0 --port 8000
Restart=always

[Install]
WantedBy=multi-user.target

sudo systemctl daemon-reload      # Перезагрузка
sudo systemctl enable myfastapi   # Запускать автоматически при загрузке ОС
sudo systemctl start myfastapi    # Запустить прямо сейчас
sudo journalctl -u myfastapi -f   # Просмотр логов
